EndlessELF writeup

2020-11-19 2 min read # Reverse

代码其实非常清楚明了。用IDA查看反编译的C代码。
发现这个程序的时间复杂度高在递归效率低下，很多数据要重复计算，于是改成循环。

#include <stdio.h>
long long key[0x1001];
char flag[25];
char byte_601050[24] = {0x6B ,0x9F,0x4E,0xA2,0x9C,0x4F,0x10,0x75,0x77,0x81,0x13,0xE6,0x9A,0x59,0x0B,0x24,0x2E,0xC5,0x42,0xB9,0xB1,0x50,0x16,0x29};

//我要的是最后一个key
int main()
{
	long long ptr[0xa0];	
	long long magic = 0xAAA,v5,v1;
	int i,j;

	for (i=0;i<=magic;i++)
	{
		if (i<=0x13)
			key[i] = i * (i + 1337) + 7331;
		else
		{
			for (j=0;j<=0x13;j++)
			{
				ptr[j] = key[i-j-1];
				if (j & 1 && j!=0x13)
					ptr[j]=ptr[j-1]-ptr[j];
			}
			
			v5=0;
			for (j=0;j<0x13;j++)
			{
				if ( j & 1 )
					v1 = ptr[j] * ptr[j];
				else
					v1 = ptr[j];
				v5 += v1;
			}
			key[i] = 2019LL*ptr[0x13]+v5;
		}
	}	
	for ( i = 0; i <= 23; ++i )
		flag[i] = *((char *)(key+0xAAA) + i % 8) ^ byte_601050[i];
	printf("ACTF{%s}\n",flag);
}

碰到的脑残问题是0xaaa+1=0x1000，也是醉了。
用到的一个好用的功能是交叉引用查找和变量重命名。右击即可。
