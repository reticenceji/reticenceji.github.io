恩尼格玛密码 Writeup

2020-08-06 2 min read # WriteUps

代码如下，读懂题目就行。
倒也可以做一个加密装置用hhh，标点符号就用V代替了。

#include <stdio.h>
#include <string.h>

struct enigma
{
	char RS;	//const
	char MK;	//change
	char rotation;	//const
	char table[30];	//const
	char rtable[30];	//const
};
struct enigma e[3]={
	{'A','B','V',"BDFHJLCPRTXVZNYEIWGAKMUSQO"},
	{'A','B','E',"AJDKSIRUXBLHWTMCQGZNPYFVOE"},
	{'A','B','Q',"EKMFLGDQVZNTOWYHXUSPAIBRCJ"},
};
char plugtable[30]="BADCFEHGJILKNMPORQTSVUXWZY";
char reverse[30]="YRUHQSLDPXNGOKMIEBFZCWVJAT";
char mima[]="AGBMCWUQCFLMXRXNIGLYVCBMQBNBYNQKXPEHJAPXOSRUIRR";


char turn(char input)
{
	char delta;	
	int i;
	int flag=0;
	input = plugtable[input-'A'];
	//转
	e[0].MK=(e[0].MK-'A'+1)%26+'A'; //1:自己会转
	e[1].MK=(e[1].MK==e[1].rotation || e[0].MK == e[0].rotation+1)?(flag=1,e[1].MK-'A'+1)%26+'A':e[1].MK;
	e[2].MK=(e[1].MK==e[1].rotation+1 && flag)?(e[2].MK-'A'+1)%26+'A':e[2].MK;
		
	for (i=0;i<3;i++)
	{
		delta = e[i].MK-e[i].RS;
		input = (e[i].table[(input-'A'+delta+26)%26]-'A'-delta+26)%26+'A';
	}
	input = reverse[input-'A'];
	for (i=2;i>=0;i--)
	{
		delta = e[i].MK-e[i].RS;
		input = (e[i].rtable[(input-'A'+delta+26)%26]-'A'-delta+26)%26+'A';
	}
	input = plugtable[input-'A'];
	return input;
}

int main()
{
	int i,j,k;
	for (i=0;i<3;i++)
		for (j=0;j<26;j++)
		{
			k=0;
			while (e[i].table[k]!=j+'A') k++;
			e[i].rtable[j]='A'+k;
		}
	
	scanf("%s",mima);
	for (i=0;i<strlen(mima);i++)
		putchar(turn(mima[i]));
	putchar('\n');
	return 0;
}
