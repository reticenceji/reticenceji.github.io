窗体(frame)：顶层窗口，即没有被包含在其他窗口中的窗口称为窗体。

package simpleframe;

//事件分配是基于awt的
import java.awt.EventQueue;
//图形绘制利用javax.swing里头的类
import javax.swing.JFrame;

public class SimpleFrame extends JFrame{
	private static final int WIDTH = 300;
	private static final int HEIGHT = 200;
	
	public SimpleFrame() {
		super.setSize(WIDTH,HEIGHT);
	}
	
	public static void main(String ...args) {
		/*所有的Swing程序组件必须由事件分派线程配置
		 * 这里简单的给关闭按钮分配了结束程序事件
		 */
		EventQueue.invokeLater(()->{
			SimpleFrame frame = new SimpleFrame();
			frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
			frame.setTitle("Hello World");
			frame.setLocation(0, 0);
			frame.setVisible(true);
		});
	}
}

package simpleframe;

import java.awt.*;
import javax.swing.*;

public class HelloWorld extends JFrame{
	//添加组件
	public HelloWorld() {
		this.add(new HelloWorldComponent());
		//java.awt.Window,自动调整窗口大小
		pack();
	}
	public static void main(String...args) {
		EventQueue.invokeLater(()->{
			JFrame frame = new HelloWorld();
			frame.setTitle("HelloWorld");
			frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
			frame.setVisible(true);
		});
	}
}

//创建一个继承了JComponent的控件
class HelloWorldComponent extends JComponent {
	public static final int MESSAGE_X = 0;
	public static final int MESSAGE_Y = 10;
	private static final int WIDTH = 300;
	private static final int HEIGHT = 200;
	
	//覆盖JComponent的这个方法，描述如何绘制控件
	public void paintComponent(Graphics g) {
        //画！
		g.drawString("Hello world",MESSAGE_X, MESSAGE_Y);
	}
	//覆盖JComponent的这个方法，描述控件大小
	public Dimension getPreferredSize() {
		return new Dimension(WIDTH,HEIGHT);
	}
}

2.5. 字体 font

创建一个字体对象，name可以使具体的字体名字也可以是逻辑字体名。style可以使Font.PLAIN,Font.BOLD,Font.ITALIC,或者是相加。
Rectangle2D getStringBounds(String s,FontRenderContext context)
返回包围这个字符串的矩形。矩形的x为基线，矩形顶端y坐标是上坡度的负值，宽度为字符串的宽度，高度为上坡度+下坡度+行间距。

示例代码：


class FontComponent extends JComponent{
	private static final int D_WIDTH = 400;
	private static final int D_HEIGHT = 400;
	
	public void paintComponent(Graphics g) {
		String message = "Hello world!!";
		Graphics2D g2 = (Graphics2D) g;
		//设置字体
		Font f = new Font("Serif",Font.BOLD,36);
		g2.setFont(f);
		//测量字体占据的矩形
		FontRenderContext context = g2.getFontRenderContext();
		Rectangle2D bounds = f.getStringBounds(message, context);
		g2.setPaint(Color.BLUE);
		g2.drawString(message,(int)(this.getWidth()-bounds.getWidth())/2,(int)(this.getHeight()-bounds.getHeight())/4);
		Shape rect = new Rectangle2D.Float(100F, 100F, 200F, 150F);
		g2.draw(rect);
		Shape ellipse = new Ellipse2D.Float(100F, 100F, 200F, 150F);
		g2.setPaint(Color.BLUE);
		g2.fill(ellipse);
		//两个端点的坐标
		Shape line = new Line2D.Float(100F, 100F, 200F+100F, 150F+100F);
		g2.draw(line);
	}
	public Dimension getPreferredSize() {
		return new Dimension(D_WIDTH,D_HEIGHT);
	}
}

3. 事件处理 Event

事件源（Event Source）：比如按钮、滚动条。
事件监听器（Event Listener）：事件源可以注册事件监听器，然后对事件做出所需的响应。
事件监听器是实现了**监听器接口（Listener interface）**的类实例。
事件对象（Event Object）：通知事件监听器发生了某个事件时，事件的相关信息会被封装在一个事件对象里。
事件对象派生于java.util.EventObject，有子类ActionEvent和WindowEvent。

事件源可以注册事件监听器————事件发生的时候，事件源将事件对象发送给所有注册的监听器————监听器在根据事件对象决定如何响应。

package simpleframe;

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class ButtonFrame extends JFrame{	
	private JPanel buttonPanel;
	private static final int D_WIDTH = 300;
	private static final int D_HEIGHT = 200;
	
	public ButtonFrame() {
		setSize(D_WIDTH,D_HEIGHT);
		//创建Button实例
		JButton button_yellow = new JButton("Yellow");
		//给Button实例添加事件监听，需要传入一个满足了ActionListenner接口的对象。
		ActionListener yellowAction = new ColorAction(Color.BLACK);
		button_yellow.addActionListener(yellowAction);
		//给窗口添加事件监听
		WindowListener listener = new Terminator();
		this.addWindowListener(listener);
		buttonPanel = new JPanel();
		buttonPanel.add(button_yellow);
		add(buttonPanel);
		
	}
	//用内部类实现接口
	private class ColorAction implements ActionListener{
		private Color backgroundcolor;
		
		public ColorAction(Color c) {
			this.backgroundcolor = c;
		}
		
		@Override
		public void actionPerformed(ActionEvent event) {
			buttonPanel.setBackground(backgroundcolor);	
		}
		
	}
	
	/*一般来说，我们可以继承WindowAdapter来获得一个空的但是满足WindowListener接口的对象
	 * 本来应该让WindowsListener接口有默认方法的，不过这是历史遗留问题了
	 * 然后我们覆盖响应的方法
	 */
	class Terminator extends WindowAdapter{
		public void windowClosing(WindowEvent e) {
			System.out.print("000");
			System.exit(0);
		}
	}
	
	public static void main(String ...arg0) {
		EventQueue.invokeLater(()->{
			JFrame frame = new ButtonFrame();
			frame.setTitle("HelloWorld");
			frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
			frame.setVisible(true);
		});
	}
}

3.2. 键盘Keyboard

动作(Action)：Action接口包含了以下方法，一般的我们继承AbstractAction类。
public void actionPerformed(ActionEvent e)
public void addPropertyChangeListener(PropertyChangeListener listener)
public Object getValue(String key)
public boolean isEnabled()
public void putValue(String key, Object value) //把一个键值对放在动作对象中。有很多键名已经有了预定义的含义。
public void removePropertyChangeListener(PropertyChangeListener listener)
public void setEnabled(boolean b)

  • 给组件添加快捷键：
    按键对象的构造：
    KeyStroke类是一个对按键描述的封装。
    KeyStroke javax.swing.KeyStroke.getKeyStroke(String description)
    根据description(由空白分隔的字符串序列，比如"ctrl B")构造一个按键对象。
    desciption格式：
    修饰符+动作+单字符
    修饰符如：shift/ctrl/alt
    动作如：typed/pressed/released
    单字符如：a,b
    键盘焦点（Keyboard Focus）：
    当用户按键时，按键事件会被发送给拥有焦点的组件。
    每个Jcomponent有三个输入映射(input map)条件，分别将KeyStroke对象映射到关联的动作。
JComponent.中的常量 条件说明
WHEN_IN_FOCUSED_WINDOW 当组件拥有键盘焦点
WHEN_FOCUSED 组件包含拥有键盘焦点的组件
WHEN_ANCESTOR_OF_FOCUSED_COMPONENT 组件和拥有键盘焦点的组件在同一个窗口

将键盘事件映射到一个组件，需要如下代码：

    //构造一个满足了Action接口的事件对象，
    Action blueAction = new ColorAction2("Blue", Color.BLUE);
    //创建一个按钮
    JButton blueButton = new JButton(blueAction);
    //创建一个按键对象
    KeyStroke ctrlBKey = KeyStroke.getKeyStroke("ctrl B");
    //将按键对象映射到按钮的InputMap中，第一个参数是输入映射条件，第二个参数是键名。
    InputMap imap = blueButton.getInputMap(JComponent.WHEN_IN_FOCUSED_WINDOW);
    imap.put(ctrlBKey, "setBlue");
    //让InputMap和ActionMap分开的目的是为了更容易让 不同输入映射 共享一个动作
    //比如：imap.put(ctrlCKey, "setBlue");
    //将Action和键名绑定在一起，完成了快捷键的注册。
    ActionMap amap = blueButton.getActionMap();
    amap.put("setBlue", blueAction);

回忆：
事件源 注册 事件监听器
一旦有事件发生，事件源 将会把 事件 发送给 注册了的所有事件监听器
根据事件，事件监听器会调用相应的 方法
下面这段无聊的代码体现这个思想：

//首先，有一个事件源button
JButton button = new JButton();
//给事件源 注册 事件监听器，这个Key就是一个满足了KeyListener接口的类。
KeyListener key = new Key();
button.addKeyListener(key);
//然后，如果我按下了一个键，就会调用这个事件监听器里头的方法KeyTyped
class Key implements KeyListener{
	public void keyPressed(KeyEvent e) {}
	public void keyReleased(KeyEvent e) {}
	public void keyTyped(KeyEvent e) {
		System.out.println(""+e.getKeyChar());	
	}
}

package dialog;

import java.awt.BorderLayout;

import javax.swing.*;

public class DialogFrame extends JFrame{
	private static final int WIDTH = 300;
	private static final int HEIGHT = 400;
	private AboutDialog dialog;
	
	public DialogFrame() {
		setSize(WIDTH,HEIGHT);
		//创建一个菜单栏
		JMenuBar menuBar = new JMenuBar();
		JMenu fileMenu = new JMenu("File");
		menuBar.add(fileMenu);
		JMenuItem aboutItem = new JMenuItem("About");
		aboutItem.addActionListener((event) -> {
			if (dialog == null) dialog = new AboutDialog(DialogFrame.this);
			dialog.setVisible(true);
		});
		fileMenu.add(aboutItem);
		JMenuItem exitItem = new JMenuItem("Exit");
		exitItem.addActionListener((event) -> {
			System.exit(0);
		});

8.3. FileChooser文件对话框

JFileChooser可以创建文件选择对话框（但不是JDialog的子类）

  1. JFileChooser使用方法如下：
    //创建一个文件选择器对象
    JFileChooser chooser = new JFileChooser();
    //设置当前的工作目录，需要提供一个File对象
    chooser.setCurrentDirectory(new File("."));
    //如果允许用户选择多个文件，设置为true，默认false
    chooser.setMultiSelectionEnabled(false);
    //默认情况下只能选择文件，可以设置选文件夹
    chooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
    //设置文件过滤器
    chooser.setFileFilter(new FileNameFilter());
    //显示对话框
    int result = chooser.showOpenDialog(this);
    //返回值显示了成功、未选择、发生错误
    if (result == JFileChooser.APPROVE_OPTION)
    {
    	//获取选择的文件，如果多个文件的话getSelectedFiles返回一个数组
    	File file = chooser.getSelectedFile();
    	System.out.println(file.getPath());
    }
  1. 文件过滤器FileFilter，一般来说就是选择文件名的后缀。创建一个文件过滤器，需要覆盖抽象类的两个抽象方法，如下所示。
private class FileNameFilter extends FileFilter{
	@Override
    //返回过滤条件，如果文件被接受，返回true
	public boolean accept(File f) {
		if (f.isDirectory()) return true;
		String name = f.getName();
		if ((name.lastIndexOf('.')<0) || !name.substring(name.indexOf('.')).equals(".java"))
			return false;
		return true;
	}

	@Override
    //返回一个过滤器的描述
	public String getDescription() {return ".java only";}
	
}
